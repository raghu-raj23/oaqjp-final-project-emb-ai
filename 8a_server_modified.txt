"""
This module initializes a Flask web application for emotion detection.

It provides endpoints to serve the main user interface page and to interact
with an underlying emotion detection package to analyze user-provided text.
"""

from flask import Flask, request, render_template
from EmotionDetection import emotion_detection

app = Flask(__name__)

@app.route("/")
def home():
    """
    Renders the default landing page.

    Returns:
        The HTML content of the main interface template.
    """
    return render_template("index.html")


@app.route("/emotionDetector")
def detect():
    """
    Handles requests to analyze the emotional tone of a given text.

    Retrieves user text from the query string parameters, processes it
    via the emotion detection service, and returns a formatted summary.

    Returns:
        str: A text response containing the score breakdown, an error
             message for invalid text, or a 400 status code error.
    """
    text_to_analyze = request.args.get("textToAnalyze")

    analysis = emotion_detection.emotion_detector(text_to_analyze)

    if analysis:
        if analysis.get('dominant_emotion') is None:
            return "<b>Invalid text! Please try again!</b>"
        response = (
            f"For the given statement, the system response is "
            f"'anger': {analysis['anger']}, 'disgust': {analysis['disgust']}, "
            f"'fear': {analysis['fear']}, 'joy': {analysis['joy']} and "
            f"'sadness': {analysis['sadness']}. "
            f"The dominant emotion is <b>{analysis['dominant_emotion']}.</b>"
        )
        return response

    return "Invalid input text.", 400

if __name__ == '__main__':
    app.run(port=5000)
